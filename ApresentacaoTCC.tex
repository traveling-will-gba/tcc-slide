\section{Introdução}

\begin{frame}{Introdução}
	\begin{itemize}[<+->]
		\item O objetivo geral deste trabalho é reescrever o jogo \textit{Traveling Will}, desenvolvido originalmente para PC na disciplina de Introdução aos Jogos Eletrônicos, para o \textit{Nintendo Game Boy Advance}
	\end{itemize}
\end{frame}

\section{Metodologia}

\begin{frame}{Implementação dos algoritmos}
	\begin{itemize}[<+->]
		\item Para este trabalho, foram implementados 3 algoritmos de multiplicação: tradicional de escola, FFT e a NTT
		\item Além disso, comparou-se suas eficiências com a biblioteca GMP do C, BigInteger do Java e a precisão arbitrária do Python
	\end{itemize}
\end{frame}

\section{Considerações Finais e Trabalhos Futuros}

\begin{frame}{Considerações Finais}
	\begin{itemize}[<+->]
		\item Com este trabalho, ficou claro que uma simples operação matemática básica como a multiplicação pode
	possuir uma análise sofisticada, envolvendo números complexos e aritmética modular
		\item Ao escolher um algoritmo de multiplicação, deve-se levar em conta os seguintes aspectos:
		\begin{enumerate}[<+->]
		 	\item quantidade de dígitos dos números de entrada
		 	\item memória disponível no hardware local
		 	\item presença e precisão da aritmética de ponto-flutuante no \textit{harwdare}
		 \end{enumerate}
	\end{itemize}
\end{frame}
