\section{Introdução}

\begin{frame}{Contextualização do problema}
	\begin{itemize}[<+->]
		\item Uso de \textit{engines} comerciais no desenvolvimento de jogos
		\item Impacto na performance dos jogos atuais
		\item Como lidar com cenários onde a performance é indispensável?
	\end{itemize}
\end{frame}

\begin{frame}{Introdução}
	\begin{itemize}[<+->]
		\item O objetivo geral deste trabalho é reescrever o jogo \textit{Traveling Will}, desenvolvido originalmente para PC na disciplina de Introdução aos Jogos Eletrônicos, para o \textit{Nintendo Game Boy Advance}
	\end{itemize}
\end{frame}

\begin{frame}{Introdução}
	Os objetivos específicos são:
	\begin{itemize}[<+->]
		\item Comprimir imagens e músicas do jogo original para reduzir o uso de memória;
		\item Criar módulos para renderização de imagens e texto;
		\item Criar módulos para manipulação de \textit{inputs} dos botões e carregamento de áudio;
		\item Criar módulos para detecção de colisões e manipulação de eventos;
		\item Criar métodos para carregamento do \textit{level design} das fases do jogo;
		\item Executar e testar o jogo desenvolvido na plataforma escolhida.
	\end{itemize}
\end{frame}

\section{Metodologia}

\begin{frame}{Ferramentas}
	Para o porte do jogo serão utilizadas as seguintes ferramentas:

		\begin{itemize}[<+->]
			\item \textit{C++}
			\item \textit{devkitARM}
			\item \textit{GIMP}
			\item \textit{VisualBoyAdvance-M}
			\item \textit{Nintendo DS}
			\item \textit{EZFlash II}
		\end{itemize}
\end{frame}


\begin{frame}{Desenvolvimento da \textit{engine}}
	\begin{itemize}[<+->]
		\item Para este trabalho será desenvolvida uma \textit{engine} que irá padronizar a utilização dos recursos providos pelo \textit{GBA};
		\item Os principais módulos a serem desenvolvidos serão: vídeo, áudio, física e \textit{input};
		\item \textbf{Módulo de vídeo}: renderizar imagens e textos, \textit{horizontal scroll};
		\item \textbf{Módulo de áudio}: iniciar, pausar e parar músicas de fundo e efeitos sonoros;
		\item \textbf{Módulo de física}: simular a ação de gravidade, detectar (opcionalmente) colisões entre objetos do jogo;
		\item \textbf{Módulo de \textit{input}}: receber o pressionamento de qualquer botão do GBA.
	\end{itemize}
\end{frame}

\begin{frame}{Reescrita do jogo}
	\begin{itemize}[<+->]
		\item Todos os recursos do jogo como imagens, áudios e arquivos de fontes precisarão ser ajustados para um formato utilizável no GBA
		\item Os principais elementos do jogo podem ser divididos em: objetos do jogo e níveis
	\end{itemize}
\end{frame}

\section{Resultados parciais}

\begin{frame}{Desenvolvimento do protótipo}
	\begin{itemize}
		\item A fim de atestar a viabilidade do porte do jogo \textit{Traveling Will} para o \textit{GBA}, foi feita uma versão funcional do menu original do jogo, já testada em um \textit{Nintendo DS};
		\item Para isso, a principal ferramenta utilizada foi a \textit{libtonc}, que nessa versão inicial fez o papel de \textit{engine} do jogo.
	\end{itemize}
\end{frame}

\begin{frame}{Desenvolvimento da \textit{gbengine}}
	\begin{itemize}
		\item Após a finalização do protótipo inicial, foi iniciado o desenvolvimento da \textit{engine} que irá substituir a \textit{libtonc} na versão final do jogo. Ela irá padronizar a utilização dos recursos providos pelo \textit{GBA};
		\item Até o momento, o módulo de \textit{input} e parte do módulo de vídeo foram implementados.
	\end{itemize}
\end{frame}

\begin{frame}{Módulo de Vídeo}
	\begin{itemize}
		\item As principais funcionalidades que o módulo de vídeo deverá ter são:
			\begin{itemize}[<+->]
				\item Escolha modo de vídeo
				\item Escolha do \textit{background} a ser utilizado
				\item Renderização de \textit{backgrounds} em qualquer região válida de memória
				\item Renderização de \textit{sprites} em qualquer região válida de memória
			\end{itemize}

	\item Funcionalidades da versão atual
		\begin{itemize}[<+->]
			\item Escolha do modo de vídeo (incompleto)
			\item Escolha do \textit{background} a ser utilizado (incompleto)
			\item Renderização de \textit{backgrounds} apenas em uma região específica de memória
		\end{itemize}
	\end{itemize}
\end{frame}
